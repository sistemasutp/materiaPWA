create database blog_db;

CREATE TABLE `users` (  
`id` int (11) AUTO_INCREMENT,  
`username`  varchar (50) DEFAULT NULL,  
`password` text (50) DEFAULT NULL,  
PRIMARY KEY (`id`)
);


CREATE TABLE `posts` (  
`id` int (11) AUTO_INCREMENT,  
`author`  integer DEFAULT NULL,  
`title` varchar (100) DEFAULT NULL,
`body` text (10000) DEFAULT NULL,  
`created` datetime DEFAULT NULL,    
PRIMARY KEY (`id`)
);


5.- configura en config.py y ejecuta python main.py
